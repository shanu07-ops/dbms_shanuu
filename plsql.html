-- 11. Write a PL/SQL block to check whether a number is palindrome using WHILE LOOP.
DECLARE
  v_num NUMBER := 121;
  v_orig NUMBER := v_num;
  v_rev NUMBER := 0;
BEGIN
  WHILE v_num > 0 LOOP
    v_rev := (v_rev * 10) + MOD(v_num, 10);
    v_num := TRUNC(v_num / 10);
  END LOOP;
  IF v_orig = v_rev THEN
    DBMS_OUTPUT.PUT_LINE('Palindrome');
  ELSE
    DBMS_OUTPUT.PUT_LINE('Not Palindrome');
  END IF;
END;

-- 12. Write a PL/SQL block to print Fibonacci series up to N terms using WHILE LOOP.
DECLARE
  n NUMBER := 10;
  a NUMBER := 0;
  b NUMBER := 1;
  c NUMBER;
  i NUMBER := 1;
BEGIN
  WHILE i <= n LOOP
    DBMS_OUTPUT.PUT(a || ' ');
    c := a + b;
    a := b;
    b := c;
    i := i + 1;
  END LOOP;
  DBMS_OUTPUT.NEW_LINE;
END;

BEGIN
  FOR i IN 2..50 LOOP
    v_is_prime := TRUE;
    FOR j IN 2..TRUNC(SQRT(i)) LOOP
      IF MOD(i, j) = 0 THEN
        v_is_prime := FALSE;
        EXIT;
      END IF;
    END LOOP;
    IF v_is_prime THEN
      DBMS_OUTPUT.PUT_LINE(i);
    END IF;
  END LOOP;
END;

-- 31. Write a stored function to return nth highest salary.
CREATE OR REPLACE FUNCTION NTH_HIGHEST(n NUMBER) RETURN NUMBER AS
  v_sal NUMBER;
BEGIN
  SELECT DISTINCT SAL INTO v_sal FROM (
    SELECT SAL, DENSE_RANK() OVER (ORDER BY SAL DESC) as rnk FROM SCOTT.EMP 
  ) WHERE rnk = n;
  RETURN v_sal;
END;

-- 32. Write a procedure to implement transaction (COMMIT / ROLLBACK).
CREATE OR REPLACE PROCEDURE TX_TEST(p_empno NUMBER, p_new_sal NUMBER) AS
BEGIN
  UPDATE SCOTT.EMP  SET SAL = p_new_sal WHERE EMPNO = p_empno;
  IF p_new_sal < 0 THEN
    ROLLBACK;
    DBMS_OUTPUT.PUT_LINE('Error: Negative Sal, Rolled Back.');
  ELSE
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Transaction Committed.');
  END IF;
END;

  BEGIN

  BEGIN

    DBMS_OUTPUT.PUT_LINE('Employee Removed');
  END;

    v_total NUMBER;
  BEGIN
    SELECT SUM(SAL) INTO v_total FROM SCOTT.EMP  WHERE DEPTNO = p_deptno;
    RETURN NVL(v_total, 0);
  END;
